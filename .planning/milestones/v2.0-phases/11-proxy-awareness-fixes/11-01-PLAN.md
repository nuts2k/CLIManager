       ```rust
       #[derive(Debug, PartialEq)]
       enum TraySwitchMode { ProxyMode, DirectMode }

       fn determine_tray_switch_mode(settings: &LocalSettings, cli_id: &str) -> TraySwitchMode {
           let global_enabled = settings.proxy.as_ref().map_or(false, |p| p.global_enabled);
           if !global_enabled { return TraySwitchMode::DirectMode; }
           let in_proxy = settings.proxy_takeover.as_ref()
               .map_or(false, |t| t.cli_ids.contains(&cli_id.to_string()));
           if in_proxy { TraySwitchMode::ProxyMode } else { TraySwitchMode::DirectMode }
       }
       ```
