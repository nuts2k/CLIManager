ProtocolType::Anthropic if path == "/v1/messages" => {
    // 判断是否有映射配置（upstream_model 或 upstream_model_map 非 None）
    let has_mapping = upstream.upstream_model.is_some()
        || upstream.upstream_model_map.is_some();
    if has_mapping {
        // 解析请求体
        let body_value: Value = serde_json::from_slice(&body_bytes)...;
        // 记录原始模型名
        let request_model = body_value.get("model")
            .and_then(|m| m.as_str())
            .unwrap_or("unknown").to_string();
        // 复用现有映射函数
        let body_value = apply_upstream_model_mapping(body_value, &upstream);
        let new_bytes = serde_json::to_vec(&body_value)...;
        let url = translate::request::build_upstream_url(&upstream.base_url, &path, &query);
        (url, Bytes::from(new_bytes), ResponseTranslationMode::AnthropicPassthrough { request_model })
    } else {
        // 无映射配置 — 纯透传（保持原有行为）
        let url = translate::request::build_upstream_url(&upstream.base_url, &path, &query);
        (url, body_bytes, ResponseTranslationMode::Passthrough)
    }

**3. 新增两个辅助函数用于响应反向映射:**

```rust
/// 非流式 Anthropic 响应中将 model 字段替换回原始请求模型名
fn reverse_model_in_response(mut body: Value, original_model: &str) -> Value {
    if let Some(obj) = body.as_object_mut() {
        if obj.contains_key("model") {
            obj.insert("model".to_string(), json!(original_model));
        }
    }
    body
}

/// 流式 SSE 行中将 model 字段值替换回原始模型名
/// 只处理 data: 开头的行中的 JSON，其他行原样透传
fn reverse_model_in_sse_line(line: &str, original_model: &str) -> String {
    if let Some(json_str) = line.strip_prefix("data: ") {
        if let Ok(mut value) = serde_json::from_str::<Value>(json_str) {
            if let Some(obj) = value.as_object_mut() {
                if obj.contains_key("model") {
                    obj.insert("model".to_string(), json!(original_model));
                    return format!("data: {}", serde_json::to_string(&value).unwrap_or_else(|_| json_str.to_string()));
                }
            }
        }
    }
    line.to_string()
}
```

**4. 新增响应分支处理 AnthropicPassthrough:**
在 `match response_mode` 中新增：
```rust
ResponseTranslationMode::AnthropicPassthrough { request_model } => {
    if !status.is_success() {
        // 4xx/5xx 直接透传
        Body::from_stream(upstream_resp.bytes_stream())
    } else if upstream_is_sse {
        // 流式：逐行扫描替换 model 字段
        // 使用与 OpenAI 分支类似的 stream wrapper 模式
        // 将上游 bytes_stream 转为按行分割的流，每行调用 reverse_model_in_sse_line
        let stream = create_anthropic_reverse_model_stream(
            upstream_resp.bytes_stream(),
            request_model,
        );
        Body::from_stream(stream)
    } else {
        // 非流式：读完整响应，替换 model 后返回
        let resp_bytes = upstream_resp.bytes().await...;
        let resp_value: Value = serde_json::from_slice(&resp_bytes)...;
        let reversed = reverse_model_in_response(resp_value, &request_model);
        let resp_bytes = serde_json::to_vec(&reversed)...;
        Body::from(resp_bytes)
    }
}
```

**5. 实现 create_anthropic_reverse_model_stream 函数:**
创建一个按行分割的 SSE 流转换器，对每个 `data: {...}` 行调用 `reverse_model_in_sse_line`。
模式参考 OpenAI 分支的 `create_anthropic_sse_stream`，但逻辑更简单 — 只做 model 字段替换，不做协议转换。
使用 `async_stream::stream!` 宏或 `futures::stream` 构建流。
内部维护一个 `buffer: String`，将收到的 bytes 追加进来，按 `\n` 分割出完整行。
对每一完整行调用 `reverse_model_in_sse_line(line, &request_model)`，yield 替换后的行 + `\n`。
缓冲区中不完整的尾行留到下一批 bytes 处理。
