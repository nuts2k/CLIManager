```rust
/// 流式请求 EOF 后回传的 token 数据
#[derive(Debug, Clone)]
pub struct StreamTokenData {
    pub input_tokens: Option<i64>,
    pub output_tokens: Option<i64>,
    pub cache_creation_tokens: Option<i64>,
    pub cache_read_tokens: Option<i64>,
    pub stop_reason: Option<String>,
}
```

2. **TrafficDb.update_streaming_log() 方法**（放在 insert_request_log 之后）：
```rust
/// 流式请求 stream 结束后更新 token/ttfb/duration 字段
///
/// 按 rowid 定位行（insert_request_log 返回的 id），
/// 仅更新非 None 字段不会将已有值覆盖为 null。
pub fn update_streaming_log(
    &self,
    id: i64,
    data: &StreamTokenData,
    ttfb_ms: Option<i64>,
    duration_ms: Option<i64>,
) -> rusqlite::Result<()> {
    let conn = self.conn.lock().unwrap();
    conn.execute(
        "UPDATE request_logs SET
            input_tokens = ?1,
            output_tokens = ?2,
            cache_creation_tokens = ?3,
            cache_read_tokens = ?4,
            stop_reason = ?5,
            ttfb_ms = ?6,
            duration_ms = ?7
         WHERE id = ?8",
        rusqlite::params![
            data.input_tokens,
            data.output_tokens,
            data.cache_creation_tokens,
            data.cache_read_tokens,
            data.stop_reason,
            ttfb_ms,
            duration_ms,
            id,
        ],
    )?;
    Ok(())
}
```

3. **测试**（放在 tests 模块末尾）：
- `test_update_streaming_log_fills_tokens`：INSERT 流式记录（token=None），UPDATE 填充，query 验证
- `test_update_streaming_log_nonexistent_id`：UPDATE 不存在的 id 返回 Ok
- `test_update_streaming_log_partial_none`：UPDATE 部分字段为 None，验证字段正确写入

注意：UPDATE 语句直接覆盖所有 7 个字段（这是设计意图——流式请求初次 INSERT 时这些字段全为 None，UPDATE 时统一设置）。
